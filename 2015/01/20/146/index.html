Array()

Array is Constructor function.

/**
* @param {...*} [args]
* @constructor
* @template T
*/
function Array(args) {}

/**
* @param {...T|Array.} [items]
* @return {Array.}
*/
Array.prototype.concat = function(items) {};

// Refer to ECMAScript,as Follows:

* @static
* @return {number}
*/
